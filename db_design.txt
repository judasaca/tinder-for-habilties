Table users {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
  pasword varchar
}
Table habilities {
  id integer
  user_id integer
}
Table transactions {
  id integer
  contract_id integer
  employee_aprove boolean
  employer_aprove boolean
  created_at timestamp
  updated_at timestamp
  status varchar [note: 'waiting-accepted-rejected']
}
Table contracts {
  id integer [primary key]
  name varchar
  employer integer [note: 'Who maeke the payments']
  employee integer [note: 'Who works']
  status varchar [note: "reviewing-accepted-rejected-finished-canceled"]
  created_at timestamp
  hourly_rate integer
  details varchar
}
